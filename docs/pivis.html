176    @staticmethod
177    def _kls(mus: np.ndarray, covs: np.ndarray) -> np.ndarray:
178        """Calculate between group KL-divergences"""
179        muq = mus.reshape(1, *mus.shape)
180        mup = mus.reshape(mus.shape[0], 1, mus.shape[1])
181        covq = covs.reshape(1, *covs.shape)
182        covp = covs.reshape(covs.shape[0], 1, covs.shape[1], covs.shape[2])
183
184        kl = -0.5 * np.log(np.linalg.det(covq) / np.linalg.det(covp)) - 0.5 * covs.shape[2] + 0.5 * np.trace(covq @ np.linalg.inv(covp), axis1=2, axis2=3) + 0.5 * ((muq - mup).reshape(mus.shape[0], mus.shape[0], 1, 2) @ np.linalg.solve(covp, (muq - mup).reshape(mus.shape[0], mus.shape[0], 2, 1))).squeeze()
185        np.fill_diagonal(kl, 0)
186        return kl

189    def from_track(cls, track: Track, threshold: float = 7.0, group_lim = 3, det_lim = 1e7, shortThreshold: int = 500, verbose: bool = True, cluster_len: int = 5) -> Self:
190        """
191        Generates AOIs from tracking information
192
193        Parameters
194        -----------
195        track : Track
196            Tracking information to use
197        threshold : float
198            Threshold for KL-divergence used to combine multiple AOIs into larger AOIs, higher is more permissive.
199            Default is 7
200        group_lim : int
201            Controls how many clusters can be rolled into 1 larger cluster in each stage of the algorithm. Higher numbers create more large groups.
202            Default is 5.
203        det_lim : float
204            Controls the maximum size of clusters via the determinant of their covariance matrices. Larger = larger clusters.
205            Default is 1e7
206        shortThreshold : int 
207            Threshold for (max timestamp - min timestamp) that this AOI is looked at, in milliseconds, AOI will be removed if below this threshold.
208            Note: This is NOT a threshold for the cumulative amount of time that the AOI is looked at.
209            Default is 500
210        verbose : bool
211            Whether or not to print out the number of AOIs remaining after each iteration.
212            Default is False.
213        cluster_len : int
214            After generating clusters based on a cleaned version of the data, 
215            when looking at the raw data, how many previous observations should be used to determine the cluster of the current observation.
216            (i.e. a smoothing factor for clustering the raw data, higher is smoother)
217            Default is 5
218        """
219        # Ignore warnings for some 'bad' things we will do
220        warnings.filterwarnings(action='ignore')
221        
222        # Initialize variables
223        mus = None
224        covs = None
225
226        data = track.df.copy()
227        time_col, IMG_X_col, IMG_Y_col, type_col, fixation_X_col, fixation_Y_col = track.cols
228        combined = True
229        prevamt = 1e99
230        while combined:
231            # Preprocessing to acquire data grouped by fixation point.
232            grouped = data.groupby([fixation_X_col, fixation_Y_col])
233            badPoints = (grouped.count()[time_col] == 1).to_numpy()
234            means = grouped[[IMG_X_col, IMG_Y_col]].mean()
235            covs = grouped[[IMG_X_col, IMG_Y_col]].cov()
236            mus = means.to_numpy()
237            covs = covs.to_numpy().reshape((mus.shape[0], 2, 2))
238            goodidx = ~(np.any(np.isnan(mus)) | badPoints)
239            index = means.index[goodidx]
240            covs = covs[goodidx] + np.repeat(np.eye(2).reshape((1, 2, 2)), np.sum(goodidx), axis=0)
241            mus = mus[goodidx]
242
243            # Used to reject if we are above the determinant limit
244            above_det_lim = lambda allFXs, allFYs: np.linalg.det(data[(data[fixation_X_col].isin(allFXs)) & (data[fixation_Y_col].isin(allFYs))][[IMG_X_col, IMG_Y_col]].cov().to_numpy()) > det_lim
245
246            # Combine clusters close enough to each other
247            kl = AreasOfInterest._kls(mus, covs)
248            groups = []
249            groupmap = {}
250            xdx, ydx = np.nonzero((kl < threshold) & (kl != 0))
251            if verbose:
252                print(len(index))
253            if len(index) < prevamt:
254                prevamt = len(index)
255            else:
256                combined = False
257                break
258
259            for x, y in zip(xdx, ydx):
260                if (x in groupmap) and (y in groupmap):
261                    pass
262                elif x in groupmap:
263                    if len(groups[groupmap[x]]) < group_lim:
264                        allFXs = [index[y][0]] + [index[z][0] for z in groups[groupmap[x]]]
265                        allFYs = [index[y][1]] + [index[z][1] for z in groups[groupmap[x]]]
266                        if above_det_lim(allFXs, allFYs):
267                            continue  
268                        groups[groupmap[x]].add(y)
269                        groupmap[y] = groupmap[x]
270                elif y in groupmap:
271                    if len(groups[groupmap[y]]) < group_lim:
272                        allFXs = [index[x][0]] + [index[z][0] for z in groups[groupmap[y]]]
273                        allFYs = [index[x][1]] + [index[z][1] for z in groups[groupmap[y]]]
274                        if above_det_lim(allFXs, allFYs):
275                            continue
276                        groups[groupmap[y]].add(x)
277                        groupmap[x] = groupmap[y]
278                elif (x not in groupmap) and (y not in groupmap):
279                    allFXs = [index[x][0], index[y][0]]
280                    allFYs = [index[x][1], index[y][1]]
281                    if above_det_lim(allFXs, allFYs):
282                        continue
283                    groups.append(set([x, y]))
284                    groupmap[x] = len(groups) - 1
285                    groupmap[y] = len(groups) - 1
286            
287            for i in range(data.shape[0]):
288                x, y = data.iloc[i][[fixation_X_col, fixation_Y_col]]
289                if not np.isnan(x):
290                    origLoc = np.nonzero(index == (x, y))[0]
291                    if len(origLoc) > 0:
292                        origLoc = origLoc[0]
293                        if goodidx[origLoc]:
294                            if origLoc in groupmap:
295                                newx, newy = index[min(groups[groupmap[origLoc]])]
296                                data.iloc[i, data.columns.get_loc(fixation_X_col)] = newx
297                                data.iloc[i, data.columns.get_loc(fixation_Y_col)] = newy
298        
299        # Revert warnings
300        warnings.filterwarnings(action='once')
301        
302        # Remove too short AOIs
303        shorts = (((grouped[time_col].max() - grouped[time_col].min())[goodidx]) < shortThreshold).to_numpy()
304        mus = mus[~shorts]
305        covs = covs[~shorts]
306
307        # Cluster points, first generate clusters based on results
308        # from cleaned data
309        data = track.raw_df
310        mvns = [stats.multivariate_normal(mus[i], covs[i]) for i in range(mus.shape[0])]
311
312        # Use the last cluster_len observations to cluster
313        toCluster = data[[IMG_X_col, IMG_Y_col]].to_numpy()[:, :, None].repeat(cluster_len, axis=2)
314        for i in range(toCluster.shape[2]):
315            toCluster[:, :, i] = np.roll(toCluster[:, :, i], i, axis=0)
316
317        # Cut off the first cluster_len - 1 observations
318        toCluster = toCluster[(cluster_len - 1):]
319        
320        # Associate points to clusters based on maximum likelihood
321        lpdfs = np.array([x.logpdf(np.moveaxis(toCluster, 1, 2)) for x in mvns])
322        lpdfs = np.sum(lpdfs, axis=2)
323        probs = special.softmax(lpdfs, axis=0).T
324        noCluster = np.all(np.isnan(probs), axis=1)
325        cluster = np.argmax(probs, axis=1)
326        
327        # Comptue transition probabilities
328        cluster = np.argmax(probs, axis=1)
329        clusterNoNa = cluster[~np.all(np.isnan(probs), axis=1)]
330        transitions = np.zeros((mus.shape[0], mus.shape[0]))
331        for i in range(1, clusterNoNa.shape[0]):
332            prev = clusterNoNa[i - 1]
333            curr = clusterNoNa[i]
334            if prev != curr:
335                transitions[prev, curr] += 1
336        transitionProbs = transitions / np.sum(transitions, axis=1).reshape((transitions.shape[0], 1))
337        
338        return cls(mus, covs, mvns, track, cluster_len, noCluster, cluster, transitionProbs)

190        """
191        Generates AOIs from tracking information
192
193        Parameters
194        -----------
195        track : Track
196            Tracking information to use
197        threshold : float
198            Threshold for KL-divergence used to combine multiple AOIs into larger AOIs, higher is more permissive.
199            Default is 7
200        group_lim : int
201            Controls how many clusters can be rolled into 1 larger cluster in each stage of the algorithm. Higher numbers create more large groups.
202            Default is 5.
203        det_lim : float
204            Controls the maximum size of clusters via the determinant of their covariance matrices. Larger = larger clusters.
205            Default is 1e7
206        shortThreshold : int 
207            Threshold for (max timestamp - min timestamp) that this AOI is looked at, in milliseconds, AOI will be removed if below this threshold.
208            Note: This is NOT a threshold for the cumulative amount of time that the AOI is looked at.
209            Default is 500
210        verbose : bool
211            Whether or not to print out the number of AOIs remaining after each iteration.
212            Default is False.
213        cluster_len : int
214            After generating clusters based on a cleaned version of the data, 
215            when looking at the raw data, how many previous observations should be used to determine the cluster of the current observation.
216            (i.e. a smoothing factor for clustering the raw data, higher is smoother)
217            Default is 5
218        """
219        # Ignore warnings for some 'bad' things we will do
220        warnings.filterwarnings(action='ignore')
221        
222        # Initialize variables
223        mus = None
224        covs = None
225
226        data = track.df.copy()
227        time_col, IMG_X_col, IMG_Y_col, type_col, fixation_X_col, fixation_Y_col = track.cols
228        combined = True
229        prevamt = 1e99
230        while combined:
231            # Preprocessing to acquire data grouped by fixation point.
232            grouped = data.groupby([fixation_X_col, fixation_Y_col])
233            badPoints = (grouped.count()[time_col] == 1).to_numpy()
234            means = grouped[[IMG_X_col, IMG_Y_col]].mean()
235            covs = grouped[[IMG_X_col, IMG_Y_col]].cov()
236            mus = means.to_numpy()
237            covs = covs.to_numpy().reshape((mus.shape[0], 2, 2))
238            goodidx = ~(np.any(np.isnan(mus)) | badPoints)
239            index = means.index[goodidx]
240            covs = covs[goodidx] + np.repeat(np.eye(2).reshape((1, 2, 2)), np.sum(goodidx), axis=0)
241            mus = mus[goodidx]
242
243            # Used to reject if we are above the determinant limit
244            above_det_lim = lambda allFXs, allFYs: np.linalg.det(data[(data[fixation_X_col].isin(allFXs)) & (data[fixation_Y_col].isin(allFYs))][[IMG_X_col, IMG_Y_col]].cov().to_numpy()) > det_lim
245
246            # Combine clusters close enough to each other
247            kl = AreasOfInterest._kls(mus, covs)
248            groups = []
249            groupmap = {}
250            xdx, ydx = np.nonzero((kl < threshold) & (kl != 0))
251            if verbose:
252                print(len(index))
253            if len(index) < prevamt:
254                prevamt = len(index)
255            else:
256                combined = False
257                break
258
259            for x, y in zip(xdx, ydx):
260                if (x in groupmap) and (y in groupmap):
261                    pass
262                elif x in groupmap:
263                    if len(groups[groupmap[x]]) < group_lim:
264                        allFXs = [index[y][0]] + [index[z][0] for z in groups[groupmap[x]]]
265                        allFYs = [index[y][1]] + [index[z][1] for z in groups[groupmap[x]]]
266                        if above_det_lim(allFXs, allFYs):
267                            continue  
268                        groups[groupmap[x]].add(y)
269                        groupmap[y] = groupmap[x]
270                elif y in groupmap:
271                    if len(groups[groupmap[y]]) < group_lim:
272                        allFXs = [index[x][0]] + [index[z][0] for z in groups[groupmap[y]]]
273                        allFYs = [index[x][1]] + [index[z][1] for z in groups[groupmap[y]]]
274                        if above_det_lim(allFXs, allFYs):
275                            continue
276                        groups[groupmap[y]].add(x)
277                        groupmap[x] = groupmap[y]
278                elif (x not in groupmap) and (y not in groupmap):
279                    allFXs = [index[x][0], index[y][0]]
280                    allFYs = [index[x][1], index[y][1]]
281                    if above_det_lim(allFXs, allFYs):
282                        continue
283                    groups.append(set([x, y]))
284                    groupmap[x] = len(groups) - 1
285                    groupmap[y] = len(groups) - 1
286            
287            for i in range(data.shape[0]):
288                x, y = data.iloc[i][[fixation_X_col, fixation_Y_col]]
289                if not np.isnan(x):
290                    origLoc = np.nonzero(index == (x, y))[0]
291                    if len(origLoc) > 0:
292                        origLoc = origLoc[0]
293                        if goodidx[origLoc]:
294                            if origLoc in groupmap:
295                                newx, newy = index[min(groups[groupmap[origLoc]])]
296                                data.iloc[i, data.columns.get_loc(fixation_X_col)] = newx
297                                data.iloc[i, data.columns.get_loc(fixation_Y_col)] = newy
298        
299        # Revert warnings
300        warnings.filterwarnings(action='once')
301        
302        # Remove too short AOIs
303        shorts = (((grouped[time_col].max() - grouped[time_col].min())[goodidx]) < shortThreshold).to_numpy()
304        mus = mus[~shorts]
305        covs = covs[~shorts]
306
307        # Cluster points, first generate clusters based on results
308        # from cleaned data
309        data = track.raw_df
310        mvns = [stats.multivariate_normal(mus[i], covs[i]) for i in range(mus.shape[0])]
311
312        # Use the last cluster_len observations to cluster
313        toCluster = data[[IMG_X_col, IMG_Y_col]].to_numpy()[:, :, None].repeat(cluster_len, axis=2)
314        for i in range(toCluster.shape[2]):
315            toCluster[:, :, i] = np.roll(toCluster[:, :, i], i, axis=0)
316
317        # Cut off the first cluster_len - 1 observations
318        toCluster = toCluster[(cluster_len - 1):]
319        
320        # Associate points to clusters based on maximum likelihood
321        lpdfs = np.array([x.logpdf(np.moveaxis(toCluster, 1, 2)) for x in mvns])
322        lpdfs = np.sum(lpdfs, axis=2)
323        probs = special.softmax(lpdfs, axis=0).T
324        noCluster = np.all(np.isnan(probs), axis=1)
325        cluster = np.argmax(probs, axis=1)
326        
327        # Comptue transition probabilities
328        cluster = np.argmax(probs, axis=1)
329        clusterNoNa = cluster[~np.all(np.isnan(probs), axis=1)]
330        transitions = np.zeros((mus.shape[0], mus.shape[0]))
331        for i in range(1, clusterNoNa.shape[0]):
332            prev = clusterNoNa[i - 1]
333            curr = clusterNoNa[i]
334            if prev != curr:
335                transitions[prev, curr] += 1
336        transitionProbs = transitions / np.sum(transitions, axis=1).reshape((transitions.shape[0], 1))
337        
338        return cls(mus, covs, mvns, track, cluster_len, noCluster, cluster, transitionProbs)
